JOBFLOW OS V2 - CLAUDE CODE BUILD INSTRUCTIONS
==============================================
You are building a full-stack job networking automation app on the user's local machine.
Read every line. Follow steps in exact order. Never skip. Never combine steps.
Run the test at the end of each step before proceeding.

=======================================
WHAT YOU ARE BUILDING
=======================================
- React + Vite frontend at localhost:5173
- FastAPI backend at localhost:8000
- WebSocket real-time communication between agents and UI
- Selenium Chrome agents that run in parallel, each working one job role
- 4-layer memory: SQLite (facts) + ChromaDB (semantic) + decision_log (WHY) + preferences (learned style)
- Two UI routes: /command (agent chat) and /dashboard (data intelligence)
- Agents brief the user proactively. User approves every message before it is sent.
- System learns user preferences across sessions. Gets smarter every run.

=======================================
ABSOLUTE RULES - NEVER BREAK THESE
=======================================
1. Every agent ALWAYS loads memory context before acting. Never acts blind.
2. Every agent ALWAYS writes results back to memory after acting. Nothing is lost.
3. Nothing is sent to LinkedIn without explicit APPROVE from user.
4. Every approve/edit/skip is recorded in decision_log with Claude-inferred WHY.
5. WebSocket is the only channel for real-time agent <-> UI communication.
6. All file outputs (stories, drafts, sent, reports) are .md or .csv files on disk.
7. config.yaml is the single source of all settings. No hardcoded values anywhere.
8. Agents explicitly tell the user which learned preferences they are applying.

  from fastapi import WebSocket
  import asyncio, json, threading

  class WebSocketManager:
    def __init__(self):
      self.connections: dict[str, WebSocket] = {}
      self.loop = None  # Set to the running event loop on startup

    async def connect(self, client_id: str, ws: WebSocket):
      await ws.accept()
      self.connections[client_id] = ws

    def disconnect(self, client_id: str):
      self.connections.pop(client_id, None)

    async def send(self, client_id: str, event_type: str, payload: dict):
      ws = self.connections.get(client_id)
      if ws:
        await ws.send_text(json.dumps({'type': event_type, 'payload': payload}))

    async def broadcast(self, event_type: str, payload: dict):
      for client_id in list(self.connections.keys()):
        await self.send(client_id, event_type, payload)

    def send_sync(self, client_id: str, event_type: str, payload: dict):
      # Thread-safe version for agent threads
      # Use asyncio.run_coroutine_threadsafe(self.send(...), self.loop)
      if self.loop:
        future = asyncio.run_coroutine_threadsafe(
          self.send(client_id, event_type, payload), self.loop
        )
        future.result(timeout=5)

  ws_manager = WebSocketManager()

            messages: [...agent.messages, { ...message, id: Date.now() }],

          }
        }
      }
    }),

    clearUnread: (agent_id) => set(state => ({

    })),

    })),

    setPendingApproval: (agent_id, approval) => set(state => ({

    })),

frontend/src/hooks/useWebSocket.js:

  import { useEffect, useRef, useCallback } from 'react'
  import { useAgentStore } from './useAgentStore'

  const WS_URL = 'ws://localhost:8000/ws'

  export function useWebSocket() {
    const ws = useRef(null)
    const store = useAgentStore()

    useEffect(() => {
      ws.current = new WebSocket(WS_URL)

      ws.current.onopen = () => console.log('WS connected')
      ws.current.onclose = () => console.log('WS disconnected')
      ws.current.onerror = (e) => console.error('WS error', e)

      ws.current.onmessage = (event) => {
        const { type, payload } = JSON.parse(event.data)
        const aid = payload?.agent_id

        switch(type) {
          case 'AGENT_BRIEFING':
          case 'AGENT_MESSAGE':
            store.addMessage(aid, { role: 'agent', text: payload.message, phase: payload.phase })
            store.upsertAgent(aid, { id: aid })
            break
          case 'APPROVAL_REQUEST':
            store.setPendingApproval(aid, payload)
            store.addMessage(aid, { role: 'approval', data: payload })
            break
          case 'AGENT_STATUS':
            store.setAgentStatus(aid, payload.status, payload.current_action)
            store.upsertAgent(aid, { company: payload.company, role: payload.role })
            break
          case 'PREFERENCE_APPLIED':
            store.addMessage(aid, { role: 'system', text: `Applying preference: ${payload.preference_rule}` })
            break
          case 'SESSION_UPDATE':
            store.setKpis(payload)
            break
          case 'PHASE_COMPLETE':
            store.addMessage(aid, { role: 'agent', text: `Phase complete. ${JSON.stringify(payload.summary_stats)}` })
            break
        }
      }

      return () => ws.current?.close()
    }, [])

    const send = useCallback((type, payload) => {
      if (ws.current?.readyState === WebSocket.OPEN) {
        ws.current.send(JSON.stringify({ type, payload }))
      }
    }, [])

    return { send }
  }

  SCORE_SYSTEM = """Score a LinkedIn profile for job role relevance.

  DRAFT_SYSTEM = """Write a LinkedIn connection invite message.

  'I hope this finds you well' or 'I'd love to', reference something real about

  def score_contact(profile: dict, context: str) -> dict:
    prompt = f"CONTEXT:\n{context[:3000]}\n\nPROFILE TO SCORE:\n{profile['name']} - {profile['title']}\n{profile.get('headline','')}\n{profile.get('summary','')[:500]}"

    return ask(prompt, system=DRAFT_SYSTEM, max_tokens=200)

  import undetected_chromedriver as uc
  from selenium.webdriver.common.by import By
  import time, random, threading
  from backend.memory.context_builder import build_agent_context
  from backend.memory.database import log_agent_action, insert_contact, update_contact_status
  from backend.memory.vector_store import embed

  from backend.websocket_manager import ws_manager
  from backend.core.claude_client import ask
  from backend.config import cfg
  import uuid

  BRIEFING_SYSTEM = """Generate a natural agent briefing message (3-5 sentences).
  Include: role being worked on, one key signal found in enrichment data,
  which preferences will be applied, what the agent plans to do next.
  Sound like a helpful team member reporting in, not a robot."""

  class LinkedInAgent:
    def __init__(self, agent_id: str, job_id: int, session_id: int):
      self.agent_id = agent_id
      self.job_id = job_id
      self.session_id = session_id
      self.context = None
      self.driver = None
      self.stop_flag = threading.Event()
      self.pause_flag = threading.Event()
      self.approval_event = threading.Event()
      self.approval_result = None  # 'approve' | 'edit' | 'skip'
      self.approval_data = None    # contains final_message if approve/edit
      self.stats = {'found': 0, 'sent': 0, 'skipped': 0}

    def emit(self, event_type: str, payload: dict):
      ws_manager.send_sync('frontend', event_type, {'agent_id': self.agent_id, **payload})

    def log(self, action: str, detail: str, status: str = 'SUCCESS'):
      log_agent_action(self.agent_id, self.job_id, self.session_id, action, detail, status)

    def start(self):
      try:
        # 1. Load context
        self.context = build_agent_context(self.job_id, self.session_id)
        self.emit('AGENT_STATUS', {'status': 'running', 'current_action': 'loading context'})

        # 2. Generate and send opening briefing
        briefing = ask(f"Write a briefing for this agent. Context summary:\n{self.context[:2000]}", system=BRIEFING_SYSTEM)
        self.emit('AGENT_BRIEFING', {'phase': 'starting', 'message': briefing})
        self.log('START', 'Agent started, context loaded')

        # 3. Launch Chrome
        options = uc.ChromeOptions()
        self.driver = uc.Chrome(options=options)
        self.emit('AGENT_STATUS', {'status': 'running', 'current_action': 'Chrome launched'})

        # 4. Navigate to LinkedIn
        self.driver.get('https://www.linkedin.com')
        time.sleep(3)
        # Check if logged in - if not, pause and notify user
        if 'feed' not in self.driver.current_url and 'login' in self.driver.current_url:
          self.emit('AGENT_MESSAGE', {'message': 'I need you to log into LinkedIn in the Chrome window I opened. Tell me when done.'})
          # Wait for user reply (handled by agent pool route_user_reply)
          self.approval_event.wait(timeout=300)
          self.approval_event.clear()

        # 5. Search people
        candidates = self.search_people()
        self.emit('AGENT_MESSAGE', {'message': f'Found {len(candidates)} candidate profiles to evaluate. Scoring now.'})

        # 6. Process each candidate
        for candidate in candidates:
          if self.stop_flag.is_set():
            break
          while self.pause_flag.is_set():
            time.sleep(1)

          profile = self.read_profile(candidate['linkedin_url'])
          if not profile:
            continue

          scored = score_contact(profile, self.context)
          if scored['score'] < cfg['agents']['relevance_min_score']:
            self.log('SCORE_LOW', f"{profile['name']} scored {scored['score']} - skipping")
            continue

          self.stats['found'] += 1
          message_draft = draft_message(profile, self.context)

          # Insert contact as pending
          contact_id = insert_contact({
            'job_id': self.job_id, 'session_id': self.session_id,
            'full_name': profile['name'], 'linkedin_url': profile['linkedin_url'],
            'title': profile['title'], 'relevance_score': scored['score'],
            'relevance_reason': scored['reason'], 'invite_message': message_draft,
            'status': 'PENDING_APPROVAL', 'agent_id': self.agent_id,
            'chroma_id': None
          })

          # Request approval
          self.request_approval(contact_id, profile, scored, message_draft)

        # 7. Session complete
        self.emit('PHASE_COMPLETE', {'phase': 'networking', 'summary_stats': self.stats})
        self.emit('AGENT_MESSAGE', {'message': f"Session complete. Found: {self.stats['found']}, Sent: {self.stats['sent']}, Skipped: {self.stats['skipped']}"})

      except Exception as e:
        self.emit('ERROR', {'error_type': 'AGENT_ERROR', 'message': str(e), 'recoverable': False})
      finally:
        if self.driver:
          self.driver.quit()

    def search_people(self) -> list[dict]:
      # Build search URL for LinkedIn people search
      # Use company name from context, filter: people, 2nd connections
      # Navigate: https://www.linkedin.com/search/results/people/?keywords={company}&network=["S"]
      # Extract profile cards: name, title, linkedin_url, mutual_connections count
      # Return list of dicts, max cfg['agents']['contacts_per_company'] * 3
      # (overshoot to account for scoring filtering)
      return []

    def read_profile(self, linkedin_url: str) -> dict | None:
      try:
        self.driver.get(linkedin_url)
        time.sleep(random.uniform(2, 3))
        self.emit('AGENT_STATUS', {'status': 'running', 'current_action': f'Reading profile'})
        # Extract: name, title, headline, about (first 300 chars), top 3 experiences
        # Return dict
        return {}
      except:
        return None

    def request_approval(self, contact_id: int, profile: dict, scored: dict, message_draft: str):
      self.approval_event.clear()
      self.approval_result = None

      context_snapshot = {
        'job_id': self.job_id, 'company': profile.get('company', ''),
        'person_title': profile['title'], 'relevance_score': scored['score'],
        'contact_id': contact_id
      }

      self.emit('APPROVAL_REQUEST', {
        'contact_id': contact_id,
        'person': {'name': profile['name'], 'title': profile['title'],
                   'linkedin_url': profile['linkedin_url'],
                   'mutual_connections': profile.get('mutual_connections', 0)},
        'message_draft': message_draft,
        'relevance': scored['score'],
        'reason': scored['reason'],
        'session_id': self.session_id,
        'context_snapshot': context_snapshot
      })
      self.emit('AGENT_STATUS', {'status': 'waiting', 'current_action': 'awaiting approval'})

      # Block until user decides (5 min timeout)
      self.approval_event.wait(timeout=300)

      if self.approval_result in ('approve', 'edit'):
        final_message = self.approval_data.get('final_message', message_draft)
        success = send_invite(self.driver, profile['linkedin_url'], final_message)
        if success:
          update_contact_status(contact_id, 'SENT')
          chroma_id = str(uuid.uuid4())
          embed('jobflow_messages', chroma_id, final_message,
                {'job_id': self.job_id, 'title': profile['title'], 'company': profile.get('company','')})
          self.stats['sent'] += 1
          self.log('SENT', f"Sent to {profile['name']}")
      elif self.approval_result == 'skip':
        self.stats['skipped'] += 1
        self.log('SKIPPED', f"Skipped {profile['name']}")

      self.emit('AGENT_STATUS', {'status': 'running', 'current_action': 'continuing'})

  import threading
  from backend.agents.linkedin_agent import LinkedInAgent
  from backend.memory.database import create_session, end_session
  from backend.websocket_manager import ws_manager
  from backend.config import cfg

  class AgentPool:
    def __init__(self):
      self.agents: dict[str, LinkedInAgent] = {}
      self.threads: dict[str, threading.Thread] = {}
      self.current_session_id: int = None

    def start_session(self, job_ids: list[int], standing_instructions: str = '') -> int:
      session_id = create_session(standing_instructions)
      self.current_session_id = session_id
      max_agents = cfg['agents']['max_parallel_agents']
      for i, job_id in enumerate(job_ids[:max_agents]):
        agent_id = f'agent_{i+1}'
        agent = LinkedInAgent(agent_id, job_id, session_id)
        if standing_instructions:
          # Inject standing instructions into agent context before start
          agent.standing_instructions = standing_instructions
        self.agents[agent_id] = agent
        t = threading.Thread(target=agent.start, daemon=True)
        self.threads[agent_id] = t
        t.start()
      return session_id

    def notify_agent(self, agent_id: str, action: str, data: dict):
      agent = self.agents.get(agent_id)
      if agent:
        agent.approval_result = action
        agent.approval_data = data
        agent.approval_event.set()

    def route_user_reply(self, agent_id: str, message: str):
      # Set a reply message on the agent so it can respond
      agent = self.agents.get(agent_id)
      if agent:
        agent.user_reply = message
        agent.approval_event.set()  # Unblock if waiting for login confirmation

    def broadcast_instruction(self, instruction: str):
      for agent in self.agents.values():
        agent.emit('AGENT_MESSAGE', {'message': f'Standing instruction received: {instruction}. Applying now.'})
        agent.standing_instructions = getattr(agent, 'standing_instructions', '') + '\n' + instruction

    def apply_standing_instruction(self, instruction: str, scope: str):
      if scope == 'ALL':
        self.broadcast_instruction(instruction)
      else:
        self.route_user_reply(scope, instruction)

    def pause_agent(self, agent_id: str):
      agent = self.agents.get(agent_id)
      if agent:
        agent.pause_flag.set()
        agent.emit('AGENT_STATUS', {'status': 'paused', 'current_action': 'paused by user'})

    def resume_agent(self, agent_id: str):
      agent = self.agents.get(agent_id)
      if agent:
        agent.pause_flag.clear()
        agent.emit('AGENT_STATUS', {'status': 'running', 'current_action': 'resumed'})

    def stop_agent(self, agent_id: str):
      agent = self.agents.get(agent_id)
      if agent:
        agent.stop_flag.set()
        agent.approval_event.set()  # Unblock if waiting

    def get_all_status(self) -> list[dict]:
      return [
        {'agent_id': aid, 'status': a.stats, 'job_id': a.job_id}
        for aid, a in self.agents.items()
      ]

    def on_all_complete(self):
      # Called when all agent threads finish
      # Import here to avoid circular import
      from backend.workers.preference_updater import run_preference_update
      if self.current_session_id:
        run_preference_update(self.current_session_id)

  pool = AgentPool()

  from backend.core.claude_client import ask_json

  import json

  UPDATE_SYSTEM = """Analyze job networking session decisions to extract preference rules.
  Given a list of decisions and their inferred reasons, identify clear durable patterns.
  Return JSON array:
  [
    {"category": "CONTACT_TARGETING", "rule": "prefers Senior ICs over recruiters", "confidence": 0.9, "evidence_count": 4},
    {"category": "TONE", "rule": "prefers direct messages under 60 words", "confidence": 0.8, "evidence_count": 3}
  ]
  Categories: CONTACT_TARGETING | TONE | THRESHOLD | TIMING | COMPANY_TYPE
  Return only JSON array. Empty array if no clear patterns."""

  def run_preference_update(session_id: int):
    decisions = get_decisions_by_session(session_id)
    if not decisions:
      end_session(session_id, {})
      return

    summary = 'SESSION DECISIONS:\n' + '\n'.join([
      f"- {d['action_type']}: {d['inferred_preference']}" 
      for d in decisions if d.get('inferred_preference')
    ])

    preferences = ask_json(summary, system=UPDATE_SYSTEM)
    updated = 0
    if isinstance(preferences, list):
      for pref in preferences:
        upsert_preference(
          pref['category'], pref['rule'],
          pref.get('evidence_count', 1), pref.get('confidence', 0.5)
        )
        updated += 1

    stats = {
      'contacts_found': sum(1 for d in decisions if d['action_type'] in ('APPROVE', 'EDIT_APPROVE', 'SKIP')),
      'messages_sent': sum(1 for d in decisions if d['action_type'] in ('APPROVE', 'EDIT_APPROVE')),
      'preferences_updated': updated
    }
    end_session(session_id, stats)

    ws_manager.send_sync('frontend', 'SESSION_COMPLETE', {
      'agent_id': 'system',
      'preferences_updated': updated,
      'summary': f'Session complete. {updated} preferences learned/updated.'
    })

  from fastapi import APIRouter
  from fastapi.responses import FileResponse, PlainTextResponse
  from backend.memory.database import get_all_jobs, get_contacts_by_job, get_session
  from backend.config import cfg
  from pathlib import Path
  import csv, datetime, io

  router = APIRouter(prefix='/api/reports', tags=['reports'])

  @router.post('/summary')
  async def generate_summary():
    jobs = get_all_jobs()
    all_contacts = []
    for job in jobs:
      contacts = get_contacts_by_job(job['id'])
      all_contacts.extend(contacts)
    sent = [c for c in all_contacts if c['status'] == 'SENT']
    lines = [
      f"# JobFlow OS Run Summary - {datetime.datetime.now().strftime('%Y-%m-%d %H:%M')}",
      f"Total roles: {len(jobs)}",
      f"Total contacts found: {len(all_contacts)}",
      f"Messages sent: {len(sent)}",
      "",
      "## Roles worked:",
    ]
    for job in jobs:
      lines.append(f"- {job['company']} / {job['role_title']} — status: {job['status']}")
    report_path = Path(cfg['paths']['reports']) / f"summary_{datetime.datetime.now().strftime('%Y%m%d_%H%M')}.md"
    report_path.parent.mkdir(exist_ok=True)
    report_path.write_text('\n'.join(lines))
    return {'file': str(report_path), 'summary': lines}

  @router.get('/contacts.csv')
  async def contacts_csv():
    jobs = get_all_jobs()
    output = io.StringIO()
    writer = csv.writer(output)
    writer.writerow(['Name', 'Title', 'Company', 'Role', 'LinkedIn', 'Score', 'Status', 'Message', 'Contacted'])
    for job in jobs:
      for c in get_contacts_by_job(job['id']):
        writer.writerow([c['full_name'], c['title'], job['company'], job['role_title'],
                         c['linkedin_url'], c['relevance_score'], c['status'],
                         c.get('invite_message',''), c.get('contacted_at','')])
    path = Path(cfg['paths']['reports']) / 'contacts.csv'
    path.write_text(output.getvalue())
    return FileResponse(path, filename='contacts.csv')
